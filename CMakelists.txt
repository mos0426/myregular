# 要求最低 CMake 版本
cmake_minimum_required(VERSION 4.3)

# 项目名称
project(MyRegular)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 如果使用 C++20 的 std::format，可改为 20
# set(CMAKE_CXX_STANDARD 20)

# 设置可执行文件的输出目录（可选）
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# 收集所有源文件
set(SOURCES
    src/internal/expression.cpp
    # 添加其他 .cpp 文件
)

# 如果只有头文件库（纯头文件实现），可以省略源文件列表，但通常至少有一个 .cpp
# Lib expression
add_library(expression STATIC ${SOURCES})
target_include_directories(expression PUBLIC
    ${PROJECT_SOURCE_DIR}/include
    ${PROJECT_SOURCE_DIR}/include/internal
)


# Unit test
add_executable(expression_test tests/internal/expression_test.cpp)
# enable_testing()
# add_test(NAME unit_test COMMAND $<TARGET_FILE:unit_test>)

target_link_libraries(expression_test PRIVATE expression GTest::gtest_main)





# 设置第三方库依赖
include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)
enable_testing()


include(GoogleTest)
gtest_discover_tests(expression_test)


# 如果使用 C++20 的 std::format，某些编译器需要链接额外的库（如 libstdc++fs 等），但通常不需要

target_compile_options(expression PRIVATE -Wall  -pedantic)

# MSVC 的警告选项
if(MSVC)
    target_compile_options(expression PRIVATE /W4)
    

endif()
